fun Fragment.requestPermissionWithRationale(
    permission: String,
    requestCode: Int,
    snackbar: AlertDialog.Builder
) {
    val provideRationale = shouldShowRequestPermissionRationale(permission)

    if (provideRationale) {
        snackbar.show()
    } else {
        requestPermissions(arrayOf(permission), requestCode)
    }
}
